public static void Shuffle(string[] a)
        {
            var rand = new Random();
            for(int i = a.Length - 1; i > 0; i--)
            {
                int j = rand.Next(0, i + 1);
                string tmp = a[i];
                a[i] = a[j];
                a[j] = tmp;
            }
        }
